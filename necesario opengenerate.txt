query para desactivar existencias 0
SELECT producto, SUM(existencia) AS total_existencia, productos.estado
FROM ExistenciaProducto inner join 
productos on productos.id = ExistenciaProducto.Producto
GROUP BY producto, productos.Estado
HAVING SUM(existencia) > 0;
